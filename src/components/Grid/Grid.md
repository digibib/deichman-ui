### Usage

```jsx
import { Grid, GridItem } from 'deichman-ui';

<Grid>
  <GridItem large="one-half">...</GridItem>
  <GridItem large="one-half">...</GridItem>
</Grid>;
```

#### Without React:

```html
<div class="grid">
  <div class="grid__item large--one-half">...</div>
  <div class="grid__item large--one-half">...</div>
</div>
```
